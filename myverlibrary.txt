class Book:
    def __init__(self, book_id, title, author, quantity):
        self.book_id = book_id
        self.title = title
        self.author = author
        self.quantity = quantity

    def is_available(self):
        return self.quantity > 0

    def borrow(self):
        if self.quantity > 0:
            self.quantity -= 1
            return True
        return False

    def return_book(self):
        self.quantity += 1

    def __str__(self):
        return f"{self.book_id} | {self.title} | {self.author} | Available: {self.quantity}"


class User:
    def __init__(self, user_id, name):
        self.user_id = user_id
        self.name = name
        self.borrowed_books = []

    def borrow_book(self, book):
        if book.borrow():
            self.borrowed_books.append(book.book_id)
            print(f"{self.name} borrowed '{book.title}'")
        else:
            print("Book not available.")

    def return_book(self, book):
        if book.book_id in self.borrowed_books:
            book.return_book()
            self.borrowed_books.remove(book.book_id)
            print(f"{self.name} returned '{book.title}'")
        else:
            print("This book was not borrowed by you.")

    def show_borrowed_books(self):
        if not self.borrowed_books:
            print("No books borrowed.")
        else:
            print("Borrowed Book IDs:", self.borrowed_books)


class Library:
    def __init__(self):
        self.books = {}
        self.users = {}

    def add_book(self, book):
        if book.book_id in self.books:
            print("Book ID already exists.")
        else:
            self.books[book.book_id] = book
            print("Book added successfully.")

    def add_user(self, user):
        if user.user_id in self.users:
            print("User ID already exists.")
        else:
            self.users[user.user_id] = user
            print("User registered successfully.")

    def display_books(self):
        if not self.books:
            print("No books in library.")
        else:
            for book in self.books.values():
                print(book)

    def search_book(self, title):
        found = False
        for book in self.books.values():
            if title.lower() in book.title.lower():
                print(book)
                found = True
        if not found:
            print("No matching book found.")

    def save_books_to_file(self):
        file = open("library_books.txt", "w")
        for book in self.books.values():
            file.write(f"{book.book_id},{book.title},{book.author},{book.quantity}\n")
        file.close()

    def load_books_from_file(self):
        try:
            file = open("library_books.txt", "r")
            for line in file:
                data = line.strip().split(",")
                book = Book(int(data[0]), data[1], data[2], int(data[3]))
                self.books[book.book_id] = book
            file.close()
        except FileNotFoundError:
            pass


def main():
    library = Library()
    library.load_books_from_file()

    while True:
        print("\n--- Library Menu ---")
        print("1. View Books")
        print("2. Add Book")
        print("3. Register User")
        print("4. Search Book")
        print("5. Borrow Book")
        print("6. Return Book")
        print("7. View Borrowed Books")
        print("8. Exit")

        choice = input("Enter choice: ")

        try:
            if choice == "1":
                library.display_books()

            elif choice == "2":
                bid = int(input("Book ID: "))
                title = input("Title: ")
                author = input("Author: ")
                qty = int(input("Quantity: "))
                library.add_book(Book(bid, title, author, qty))

            elif choice == "3":
                uid = int(input("User ID: "))
                name = input("Name: ")
                library.add_user(User(uid, name))

            elif choice == "4":
                title = input("Enter title to search: ")
                library.search_book(title)

            elif choice == "5":
                uid = int(input("User ID: "))
                bid = int(input("Book ID: "))
                user = library.users.get(uid)
                book = library.books.get(bid)
                if user and book:
                    user.borrow_book(book)
                else:
                    print("Invalid user or book.")

            elif choice == "6":
                uid = int(input("User ID: "))
                bid = int(input("Book ID: "))
                user = library.users.get(uid)
                book = library.books.get(bid)
                if user and book:
                    user.return_book(book)
                else:
                    print("Invalid user or book.")

            elif choice == "7":
                uid = int(input("User ID: "))
                user = library.users.get(uid)
                if user:
                    user.show_borrowed_books()
                else:
                    print("Invalid user.")

            elif choice == "8":
                library.save_books_to_file()
                print("Data saved. Exiting...")
                break

            else:
                print("Invalid choice.")

        except ValueError:
            print("Please enter valid numeric input.")


if __name__ == "__main__":
    main()
